\section{Problem and System}

\begin{frame}[shrink=20]{Business Problem and Stakeholders}
  \begin{block}{Problem}
    Users need answers to time-aware factual questions, but Freebase requires schema knowledge, entity IDs, relation IDs, and SPARQL expertise~\parencite{bollacker2008freebase}.
  \end{block}
  \begin{columns}[T]
    \column{0.52\textwidth}
    \begin{exampleblock}{Example questions}
      \small
      \begin{itemize}
        \item Who was US president before Obama?
        \item What was the first Harry Potter novel?
        \item Where did an artist live at a given time?
      \end{itemize}
    \end{exampleblock}
    \column{0.44\textwidth}
    \begin{block}{Stakeholders}
      \small
      \begin{itemize}
        \item analysts and researchers
        \item BI and compliance teams
        \item knowledge-graph engineers
      \end{itemize}
    \end{block}
  \end{columns}
  \vspace{0.15cm}
  \begin{alertblock}{Submission framing}
    This is a production-oriented NLP system artifact, not a research-only notebook.
  \end{alertblock}
\end{frame}

\begin{frame}[shrink=20]{Why Temporal KBQA Needs NLP}
  \begin{columns}[T]
    \column{0.53\textwidth}
    \begin{block}{NLP task}
      Map free-form temporal language into executable KB structure:
      \begin{center}
        \scriptsize\texttt{Question -> S-expression -> SPARQL -> Answer}
      \end{center}
    \end{block}
    \begin{exampleblock}{Temporal signals}
      \small before, after, during, first, last, earliest, latest, now
    \end{exampleblock}
    \column{0.43\textwidth}
    \begin{block}{Success metrics}
      \small
      \begin{itemize}
        \item F1 / Hits@1 / Accuracy
        \item valid S-expression rate
        \item answer rate and executor errors
        \item demo/API readiness
      \end{itemize}
    \end{block}
  \end{columns}
  \vspace{0.15cm}
  \begin{alertblock}{Current result framing}
    The v1 system is submission-ready as an engineering artifact, but not performance-ready as a production QA model.
  \end{alertblock}
\end{frame}
